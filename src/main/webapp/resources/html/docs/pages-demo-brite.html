Charts using Brite Charts

Britecharts is a client-side reusable Charting Library based on D3.js v5 that offers easy and intuitive use of charts and components that can be composed together to create amazing visualizations.

Following is simple example of line and bar chart, but you can create many different types of charts very easily. Since this is based on D3, it opens up door to create charts with many advance options. You can refer to official documentation here. Also checkout js file demo.britechart.js file for more example.

                                                                function createBarChart(elementSelector, data) {
                                                                    var chartTooltip = new britecharts.miniTooltip();
                                                                    var _barChart = new britecharts.bar();
                                                                    var barContainer = d3.select(elementSelector),
                                                                        containerWidth = barContainer.node()
                                                                            ? barContainer.node().getBoundingClientRect().width
                                                                            : false,
                                                                        containerHeight = barContainer.node()
                                                                            ? barContainer.node().getBoundingClientRect().height
                                                                            : false,
                                                                        tooltipContainer = void 0;
                                                        
                                                                    var margin = {
                                                                        top: 10,
                                                                        left: 55,
                                                                        bottom: 20,
                                                                        right: 10
                                                                    };
                                                        
                                                                    _barChart
                                                                        .isAnimated(true)
                                                                        .width(containerWidth)
                                                                        .height(containerHeight)
                                                                        .margin(margin)
                                                                        .betweenBarsPadding(0.5)
                                                                        .colorSchema(["#727cf5"])
                                                                        .on("customMouseOver", chartTooltip.show)
                                                                        .on("customMouseMove", chartTooltip.update)
                                                                        .on("customMouseOut", chartTooltip.hide);
                                                        
                                                                    barContainer.datum(data).call(_barChart);
                                                                    tooltipContainer = d3.select(elementSelector + " .metadata-group");
                                                                    tooltipContainer.datum([]).call(chartTooltip);
                                                                }
                                                        
                                                                
                                                                function createLineChart(elementSelector, data) {
                                                                    var lineChart = new britecharts.line(),
                                                                        chartTooltip = new britecharts.tooltip(),
                                                                        lineContainer = d3.select(elementSelector),
                                                                        containerWidth = lineContainer.node()
                                                                            ? lineContainer.node().getBoundingClientRect().width
                                                                            : false,
                                                                        tooltipContainer = void 0;
                                                        
                                                                    lineChart
                                                                        .isAnimated(true)
                                                                        .aspectRatio(0.7)
                                                                        .tooltipThreshold(100)
                                                                        .grid("horizontal")
                                                                        .width(containerWidth)
                                                                        .dateLabel("date")
                                                                        .valueLabel("value")
                                                                        .on("customMouseOver", chartTooltip.show)
                                                                        .on("customMouseMove", chartTooltip.update)
                                                                        .on("customMouseOut", chartTooltip.hide);
                                                        
                                                                    chartTooltip.title("Sample Data");
                                                        
                                                                    
                                                        
                                                                    lineContainer.datum(data).call(lineChart);
                                                                    tooltipContainer = d3.select(
                                                                        elementSelector + " .metadata-group .hover-marker"
                                                                    );
                                                                    tooltipContainer.datum([]).call(chartTooltip);
                                                                };

                                                                function drawCharts() {
                                                                    d3.selectAll('.bar-chart').remove();
                                                                    d3.selectAll('.line-chart').remove();
                                                        
                                                        
                                                                    if ($(".bar-container").length > 0) {
                                                                        createBarChart(".bar-container", barData);
                                                                    }
                                                        
                                                                    if ($(".line-container").length > 0) {
                                                                        createLineChart(".line-container", lineData);
                                                                    }
                                                                }
                                                        
                                                                $(window).on('resize', function (e) {
                                                                    e.preventDefault();
                                                                    setTimeout(drawCharts, 200);
                                                                });
                                                        
                                                        
                                                                drawCharts();
